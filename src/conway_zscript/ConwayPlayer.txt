class ConwayPlayer: DoomPlayer
{
	Default {
		+NOGRAVITY;
		Player.WeaponSlot 1, "ConwaySimulationGun";
		Player.WeaponSlot 2, "Pistol";
		Player.WeaponSlot 3, "Shotgun";
	}
	override void PostBeginPlay()
	{
		super.PostBeginPlay();
		self.bFly = true;
		A_Log("ConwayPlayer init");
		A_GiveInventory("ConwaySimulationGun", 1);
	}
}
