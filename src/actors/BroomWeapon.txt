actor BroomWeapon: Chainsaw 20103
{

	Weapon.BobStyle Smooth
	Weapon.BobRangeX 0.5
	Weapon.BobRangeY 0.2
	
	Weapon.UpSound 0
	Weapon.ReadySound 0
	
	Inventory.PickupMessage "Time to start sweeping!"
	Tag "Broom"
	+WEAPON.MELEEWEAPON
	
	scale .2
	
	States
	{
		Ready:
		DBWG A 1 A_WeaponReady
		loop
		
		Select:
		DBWG A 1 A_Raise
		loop
		
		Deselect:
		DBWG A 1 A_LOWER
		loop
		
		Fire:
		DBWF AB 4 A_Saw("weapon/broomsweep", "weapon/broomsweep")
		DBWF B 0 A_ReFire
		goto Ready
		
		Spawn:
		DBWP A -1
		stop
	}
}
