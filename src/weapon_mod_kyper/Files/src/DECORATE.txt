//using 20100 -weapons
actor ImpGun: Weapon Replaces Pistol 20100
{
	Weapon.SlotNumber 2
	Weapon.SlotPriority 1
	Weapon.SelectionOrder 50
	Weapon.AmmoUse 5
	Weapon.AmmoGive 100
	Weapon.AmmoType "Clip"
	Weapon.BobStyle Smooth
	Weapon.BobRangeX 0.4
	Weapon.BobRangeY 0.2
	
	
	Scale .5
	
	Inventory.PickupSound "weapon/impfire"
	Inventory.PickupMessage "IMP GUN"
	
	States
	{
		Spawn:
		IHFS A -1
		Stop
		
		Select:
		IHFG A 1 A_Raise
		Loop
		
		Deselect:
		IHFG A 1 A_Lower
		Loop
		
		Ready:
		IHFG A 1 A_WeaponReady
		Loop
		
		Fire:
		IHFF A 4
		Goto ImpFire
		
		ImpFire:
		IHFF A 0 A_Playsound("DSHOOF",CHAN_WEAPON,0.7)
		IHFF A 0 A_Recoil(4)
		IHFF A 5 A_FireCustomMissile("IMPFIRE_MISSILE",0,0,0,-8)
		IHFF A 5 A_Refire("ImpFire")
		Goto Ready
	}
	
}
